import http.client
import json
import openpyxl

# Function to get sentiment from API
def get_sentiment(feedback):
    conn = http.client.HTTPSConnection("openai-nucleus-dev.azpriv-cloud.ubs.net")

    payload = json.dumps({
        "referenceId": "",
        "labels": [],
        "serviceId": "1f2a5db1_ee85_49e9_ac7e_71813efcf574",
        "text": feedback,
        "documentName": "sa_api1.txt",
        "etag": "",
        "source": "test.txt",
        "mimeType": "txt",
        "clientMetadata": {
            "additionalProp1": {},
            "additionalProp2": {},
            "additionalProp3": {}
        }
    })

    headers = {
        'Authorization': 'Bearer 221afc03-c0d2-4979-8914-22fa46ba9e05',
        'Content-Type': 'application/json'
    }

    conn.request("POST", "/api/v1/processing-document-service/organizations/test_gf_hr_ah_dev/workflow-documents/_string/results", payload, headers)
    res = conn.getresponse()
    data = res.read().decode("utf-8")
    conn.close()

    json_obj = json.loads(data)
    json_body = json_obj[0]
    sentiment = json_body['enrichment'][0]['textEnrichment']['values'][0]['value']
    return sentiment

# Function to process the Excel file
def process_excel(input_path, output_path):
    workbook = openpyxl.load_workbook(input_path)
    sheet = workbook.active

    for row in sheet.iter_rows(min_row=2, max_row=sheet.max_row, min_col=1, max_col=1):
        feedback = row[0].value
        sentiment = get_sentiment(feedback)
        sheet.cell(row=row[0].row, column=2, value=sentiment)

    workbook.save(output_path)

# Example usage
input_path = '/mnt/data/your_input.xlsx'
output_path = '/mnt/data/your_output.xlsx'
process_excel(input_path, output_path)
